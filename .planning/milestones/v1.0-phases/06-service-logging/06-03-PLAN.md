Create `tests/test-phase6.sh` following the established test pattern from existing phase test scripts. The test must:

1. **Setup:** Create a temporary log directory, export LOG_DIR to it, export all LOG_* vars as 1, start services with `docker compose up -d`.

2. **LOG-01 (Hook logging):** Trigger a hook execution by running a curl command inside the claude container via `docker compose exec`. Check that `$LOG_DIR/hook.jsonl` exists and contains at least one line.

3. **LOG-02 (Proxy logging):** Make an HTTP request through the proxy (e.g., `docker compose exec claude curl -s http://proxy:8080/v1/messages -X POST -H 'Content-Type: application/json' -d '{"model":"test","messages":[]}'`). Check that `$LOG_DIR/anthropic.jsonl` exists and contains at least one line.

4. **LOG-03 (Validator logging):** The validator logs on startup (iptables setup, DB init). Check that `$LOG_DIR/iptables.jsonl` exists and contains at least one line.

5. **LOG-04 (Unified directory):** Verify all three JSONL files exist in the same `$LOG_DIR` directory: `ls $LOG_DIR/hook.jsonl $LOG_DIR/anthropic.jsonl $LOG_DIR/iptables.jsonl`.

6. **LOG-05 (JSON structure):** For each JSONL file, read the first line and validate it has ALL FOUR required fields: `ts`, `svc`, `level`, AND `msg` using `jq -e '.ts and .svc and .level and .msg'`. This check MUST include `.msg` -- omitting it would allow LOG-05 non-compliant entries to pass.

7. **LOG-06 (Disabled by default):** Stop services. Create a new temp log dir. Export all LOG_* vars as 0, restart services, trigger a proxy request, wait 2 seconds, verify no JSONL files were created in the new log dir.

8. **LOG-07 (Logs command):** Verify `bin/claude-secure` contains the `logs)` case. (Full functional test of `tail -f` is not practical in CI since it blocks, so verify the code path exists and the clear subcommand works by creating a dummy file and running the clear logic.)

**Test structure:**

```bash
#!/bin/bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
PASS=0
FAIL=0
TOTAL=0

pass() { ((PASS++)); ((TOTAL++)); echo "  PASS: $1"; }
fail() { ((FAIL++)); ((TOTAL++)); echo "  FAIL: $1"; }

cleanup() {
  cd "$PROJECT_DIR"
  LOG_HOOK=0 LOG_ANTHROPIC=0 LOG_IPTABLES=0 docker compose down 2>/dev/null || true
  rm -rf "$TEST_LOG_DIR" "$TEST_LOG_DIR_DISABLED" 2>/dev/null || true
}
trap cleanup EXIT

# Create test log directory
TEST_LOG_DIR=$(mktemp -d)
TEST_LOG_DIR_DISABLED=$(mktemp -d)
chmod 777 "$TEST_LOG_DIR" "$TEST_LOG_DIR_DISABLED"

cd "$PROJECT_DIR"

echo "=== Phase 6: Service Logging Tests ==="
echo ""

# Start services with logging enabled
echo "Starting services with logging enabled..."
export LOG_DIR="$TEST_LOG_DIR"
export LOG_HOOK=1 LOG_ANTHROPIC=1 LOG_IPTABLES=1
docker compose up -d
sleep 5  # Wait for services to initialize

# LOG-01: Hook logging
echo "LOG-01: Hook writes JSON log when LOG_HOOK=1"
docker compose exec -T claude bash -c 'echo "{\"tool_name\":\"Bash\",\"tool_input\":{\"command\":\"echo hello\"}}" | bash /etc/claude-secure/hooks/pre-tool-use.sh' >/dev/null 2>&1 || true
sleep 1
if [ -f "$TEST_LOG_DIR/hook.jsonl" ] && [ -s "$TEST_LOG_DIR/hook.jsonl" ]; then
  pass "LOG-01: hook.jsonl exists and is non-empty"
else
  fail "LOG-01: hook.jsonl missing or empty"
fi

# LOG-02: Proxy logging
echo "LOG-02: Proxy writes JSON log when LOG_ANTHROPIC=1"
docker compose exec -T claude curl -s http://proxy:8080/v1/messages -X POST -H 'Content-Type: application/json' -d '{"model":"test","messages":[]}' >/dev/null 2>&1 || true
sleep 1
if [ -f "$TEST_LOG_DIR/anthropic.jsonl" ] && [ -s "$TEST_LOG_DIR/anthropic.jsonl" ]; then
  pass "LOG-02: anthropic.jsonl exists and is non-empty"
else
  fail "LOG-02: anthropic.jsonl missing or empty"
fi

# LOG-03: Validator logging
echo "LOG-03: Validator writes JSON log when LOG_IPTABLES=1"
if [ -f "$TEST_LOG_DIR/iptables.jsonl" ] && [ -s "$TEST_LOG_DIR/iptables.jsonl" ]; then
  pass "LOG-03: iptables.jsonl exists and is non-empty"
else
  fail "LOG-03: iptables.jsonl missing or empty"
fi

# LOG-04: Unified directory
echo "LOG-04: All logs in unified host directory"
if [ -f "$TEST_LOG_DIR/hook.jsonl" ] && [ -f "$TEST_LOG_DIR/anthropic.jsonl" ] && [ -f "$TEST_LOG_DIR/iptables.jsonl" ]; then
  pass "LOG-04: all three JSONL files in $TEST_LOG_DIR"
else
  fail "LOG-04: not all JSONL files present"
fi

# LOG-05: JSON structure validation — MUST check all four fields: ts, svc, level, msg
echo "LOG-05: Log entries have ts, svc, level, msg fields"
LOG05_PASS=true
for f in hook.jsonl anthropic.jsonl iptables.jsonl; do
  if [ -f "$TEST_LOG_DIR/$f" ]; then
    if ! head -1 "$TEST_LOG_DIR/$f" | jq -e '.ts and .svc and .level and .msg' >/dev/null 2>&1; then
      LOG05_PASS=false
      fail "LOG-05: $f first line missing required fields (need ts, svc, level, msg)"
    fi
  fi
done
if [ "$LOG05_PASS" = "true" ]; then
  pass "LOG-05: all log entries have required JSON fields (ts, svc, level, msg)"
fi

# Stop for disabled test
docker compose down 2>/dev/null || true

# LOG-06: Disabled by default
echo "LOG-06: No logs when flags are 0"
export LOG_DIR="$TEST_LOG_DIR_DISABLED"
export LOG_HOOK=0 LOG_ANTHROPIC=0 LOG_IPTABLES=0
docker compose up -d
sleep 5
docker compose exec -T claude curl -s http://proxy:8080/v1/messages -X POST -H 'Content-Type: application/json' -d '{"model":"test","messages":[]}' >/dev/null 2>&1 || true
sleep 2
FOUND_LOGS=false
for f in hook.jsonl anthropic.jsonl iptables.jsonl; do
  if [ -f "$TEST_LOG_DIR_DISABLED/$f" ] && [ -s "$TEST_LOG_DIR_DISABLED/$f" ]; then
    FOUND_LOGS=true
  fi
done
if [ "$FOUND_LOGS" = "false" ]; then
  pass "LOG-06: no JSONL files created when logging disabled"
else
  fail "LOG-06: JSONL files found when logging should be disabled"
fi

docker compose down 2>/dev/null || true

# LOG-07: logs command exists
echo "LOG-07: claude-secure logs command"
if grep -q 'logs)' "$PROJECT_DIR/bin/claude-secure" && grep -q 'tail -f' "$PROJECT_DIR/bin/claude-secure"; then
  pass "LOG-07: logs subcommand exists in CLI with tail -f"
else
  fail "LOG-07: logs subcommand missing from CLI"
fi

# Summary
echo ""
echo "=== Results: $PASS/$TOTAL passed, $FAIL failed ==="
[ "$FAIL" -eq 0 ] && exit 0 || exit 1
```
