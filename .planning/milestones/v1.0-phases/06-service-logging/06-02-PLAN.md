# Parse log flags from arguments, export env vars for docker compose
parse_log_flags() {
  LOG_HOOK=0
  LOG_ANTHROPIC=0
  LOG_IPTABLES=0
  REMAINING_ARGS=()

  for arg in "$@"; do
    case "$arg" in
      log:hook)       LOG_HOOK=1 ;;
      log:anthropic)  LOG_ANTHROPIC=1 ;;
      log:iptables)   LOG_IPTABLES=1 ;;
      log:all)        LOG_HOOK=1; LOG_ANTHROPIC=1; LOG_IPTABLES=1 ;;
      *)              REMAINING_ARGS+=("$arg") ;;
    esac
  done

  export LOG_HOOK LOG_ANTHROPIC LOG_IPTABLES
}
```

  logs)
    LOG_SVC="${REMAINING_ARGS[1]:-}"
    case "$LOG_SVC" in
      hook)       tail -f "$LOG_DIR/hook.jsonl" 2>/dev/null ;;
      anthropic)  tail -f "$LOG_DIR/anthropic.jsonl" 2>/dev/null ;;
      iptables)   tail -f "$LOG_DIR/iptables.jsonl" 2>/dev/null ;;
      clear)
        rm -f "$LOG_DIR"/*.jsonl
        echo "Logs cleared."
        ;;
      "")
        # Guard against unmatched glob when no .jsonl files exist yet
        shopt -s nullglob
        LOG_FILES=("$LOG_DIR"/*.jsonl)
        shopt -u nullglob
        if [ ${#LOG_FILES[@]} -eq 0 ]; then
          echo "No log files found in $LOG_DIR. Start services with log:* flags first."
          exit 0
        fi
        tail -f "${LOG_FILES[@]}"
        ;;
      *)
        echo "Unknown log service: $LOG_SVC"
        echo "Usage: claude-secure logs [hook|anthropic|iptables|clear]"
        exit 1
        ;;
    esac
    ;;
```

In the `setup_directories()` function (line 90), add log directory creation after the existing `mkdir -p "$CONFIG_DIR"`:

```bash
setup_directories() {
  mkdir -p "$CONFIG_DIR"
  chmod 700 "$CONFIG_DIR"
  log_info "Created config directory: $CONFIG_DIR"

  # Create logs directory for service logging (LOG_DIR in docker-compose)
  # chmod 755: owner has full access, others can read/traverse.
  # Container processes write via the bind mount where Docker maps UIDs.
  # If container UID mismatch causes write failures, the CLI wrapper's
  # mkdir -p at launch time can adjust permissions as needed.
  mkdir -p "$CONFIG_DIR/logs"
  chmod 755 "$CONFIG_DIR/logs"
  log_info "Created logs directory: $CONFIG_DIR/logs"
}
```

Note: Using `chmod 755` (not 777) to maintain tighter permissions consistent with the project's security posture. The logs directory only contains JSONL log files with no secrets, but world-writable directories are unnecessary since Docker bind mounts handle UID mapping. If specific container UIDs need write access, the user can adjust after installation or the CLI wrapper can handle it.
