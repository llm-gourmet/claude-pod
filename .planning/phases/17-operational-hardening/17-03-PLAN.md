    ```bash
    scenario_concurrent_execution() {
      local secret="e2e-test-secret"
      local body='{"action":"opened","issue":{"title":"e2e-concurrent","number":2},"repository":{"full_name":"e2e/test"}}'
      local sig
      sig=$(printf '%s' "$body" | openssl dgst -sha256 -hmac "$secret" -binary | xxd -p -c256)

      for i in 1 2 3; do
        curl -sS -X POST "http://127.0.0.1:$LISTENER_PORT/webhook" \
          -H "X-Hub-Signature-256: sha256=$sig" \
          -H 'X-GitHub-Event: issues' \
          -H "X-GitHub-Delivery: e2e-concurrent-$i" \
          -d "$body" >/dev/null &
      done
      wait

      # Poll for 3 audit lines, bounded by 60s inner deadline
      local jsonl="$CONFIG_DIR/logs/cs-e2e-executions.jsonl"
      local deadline=$((SECONDS + 60))
      while [ "$SECONDS" -lt "$deadline" ]; do
        local n
        n=$(wc -l < "$jsonl" 2>/dev/null || echo 0)
        [ "$n" -ge 3 ] && break
        sleep 1
      done

      local audit_count
      audit_count=$(wc -l < "$jsonl" 2>/dev/null || echo 0)
      [ "$audit_count" -eq 3 ] || { echo "FAIL concurrent: $audit_count audit lines (expected 3)"; return 1; }

      # Every line must be jq-parseable (no JSONL corruption from concurrent writes)
      if ! jq -c . < "$jsonl" >/dev/null 2>&1; then
        echo "FAIL concurrent: corrupt JSONL line detected"
        return 1
      fi

      # Bare repo must have received 3 report pushes (initial + 3 = >= 4 commits on main)
      local bare="$TEST_TMPDIR/e2e-reports.git"
      local commits
      commits=$(git -C "$bare" rev-list --count main 2>/dev/null || echo 0)
      [ "$commits" -ge 4 ] || { echo "FAIL concurrent: report repo has $commits commits (expected >=4)"; return 1; }

      return 0
    }
    ```

    ```bash
    scenario_resource_limits() {
      # Find a claude service container in the cs-e2e- namespace.

      local claude_cid

                    | head -1)
      if [ -z "$claude_cid" ]; then
        # Fallback: any recent claude service container matching the prefix via project label
        claude_cid=$(docker ps -aq --filter "label=com.docker.compose.service=claude" | head -1)

      mem_bytes=$(docker inspect --format '{{.HostConfig.Memory}}' "$claude_cid")
    ```
