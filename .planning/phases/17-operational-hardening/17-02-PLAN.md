reap_orphan_projects() {
  # Arg: $1 = prefix (default from INSTANCE_PREFIX, else "cs-")
  # Walks docker ps -a --filter "label=com.docker.compose.project",
  # post-filters by prefix + cs-*-* shape, looks up Created via docker inspect,
  # compares age to REAPER_ORPHAN_AGE_SECS (default 600), invokes
  # `docker compose -p <proj> down -v --remove-orphans --timeout 10` per match.
  # Sets globals: REAPED_COUNT, REAPED_ERRORS.
  # In --dry-run mode, prints the kill list and skips the actual down.
}

reap_stale_event_files() {
  # Walks $CONFIG_DIR/events/*.json with `find -mmin +$((REAPER_EVENT_AGE_SECS/60))`,
  # counts deletions into global EVENTS_DELETED.
  # Default threshold: REAPER_EVENT_AGE_SECS=86400 (24h).
  # In --dry-run mode, prints the sweep list and skips the actual rm.
}

do_reap() {
  # Args: optional --dry-run flag.
  # Flow:
  #   1. Parse flags (--dry-run sets REAPER_DRY_RUN=1)
  #   2. mkdir -p $(dirname lock_file); exec 9>$lock_file; flock -n 9 (D-08)
  #   3. echo "reaper: cycle start prefix=$INSTANCE_PREFIX"
  #   4. REAPED_COUNT=0 REAPED_ERRORS=0 EVENTS_DELETED=0
  #   5. reap_orphan_projects "$INSTANCE_PREFIX"
  #   6. reap_stale_event_files
  #   7. echo "reaper: cycle end killed=$REAPED_COUNT events_deleted=$EVENTS_DELETED errors=$REAPED_ERRORS"
  #   8. return nonzero only if whole-cycle failure (REAPED_ERRORS > 0 && REAPED_COUNT == 0)
}
