**Script header:**
```bash
#!/bin/bash
# pre-tool-use.sh -- PreToolUse hook for claude-secure call validation
# Intercepts Bash/WebFetch/WebSearch tool calls, checks domain whitelist,
# registers call-IDs with validator for allowed payload calls.
set -euo pipefail
```

**Constants:**
```bash
WHITELIST="/etc/claude-secure/whitelist.json"
VALIDATOR_URL="http://127.0.0.1:8088"
LOG_FILE="/var/log/claude-secure/hook.log"
```

**Capture stdin immediately** (single-read stream, per Pitfall 3 in research):
```bash
INPUT=$(cat)
```

**log() function:** `echo "[$(date -Iseconds)] $*" >> "$LOG_FILE" 2>/dev/null || true`

**deny() function:** Takes a reason string. Outputs the JSON deny response on stdout and exits 0:
```bash
deny() {
  local reason="$1"
  log "DENY: $reason"
  jq -n --arg reason "$reason" '{
    hookSpecificOutput: {
      hookEventName: "PreToolUse",
      permissionDecision: "deny",
      permissionDecisionReason: $reason
    }
  }'
  exit 0
}
```
NOTE: Per research Pitfall 6 and the verified Claude Code protocol, use exit 0 with JSON `permissionDecision: "deny"` -- NOT exit 2 with stderr. This ensures Claude sees the structured reason. This supersedes D-01's mention of exit 2.

**allow() function:** Logs and exits 0 with no stdout output (bare exit 0 = allow):
```bash
allow() {
  log "ALLOW: $1"
  exit 0
}
```

**Extract tool name and input:**
```bash
TOOL_NAME=$(echo "$INPUT" | jq -r '.tool_name // empty')
```

**extract_domain() function:** Returns the target domain (or empty string) based on tool type.

For **Bash** tool:
1. Get command: `cmd=$(echo "$INPUT" | jq -r '.tool_input.command // empty')`
2. Check if command contains `curl` or `wget`: `if echo "$cmd" | grep -qE '\bcurl\b|\bwget\b'; then`
3. Check for shell obfuscation indicators that could hide the real URL -- if the curl/wget portion contains `$`, backticks, `eval`, `base64` in the URL context, deny (per D-04 fail-closed and Pitfall 7):
   ```bash
   # Extract the part of the command from curl/wget onward
   local http_cmd
   http_cmd=$(echo "$cmd" | grep -oE '(curl|wget)\s+.*')
   if echo "$http_cmd" | grep -qE '(\$[{(]|\$[A-Za-z]|`|eval\s|base64)'; then
     echo "__OBFUSCATED__"
     return
   fi
   ```
4. Extract URL with regex: `echo "$cmd" | grep -oP 'https?://[^\s"'\''()<>|;&]+' | head -1`
5. Extract domain from URL: `echo "$url" | sed -E 's|https?://([^/:]+).*|\1|'`
6. If no URL found but curl/wget is present, return `__NO_URL__` (fail-closed)
7. If command does NOT contain curl/wget, return empty (non-network Bash command, allow)

For **WebFetch** tool:
1. `echo "$INPUT" | jq -r '.tool_input.url // empty' | sed -E 's|https?://([^/:]+).*|\1|'`

For **WebSearch** tool:
1. Return empty string (WebSearch is a search query with no specific target domain -- treated as read-only, allowed per CALL-04)

**has_payload() function:** Returns 0 (true) if the Bash command has outbound payload indicators, 1 (false) otherwise.
```bash
has_payload() {
  local cmd
  cmd=$(echo "$INPUT" | jq -r '.tool_input.command // empty')
  # HTTP method flags indicating non-GET
  echo "$cmd" | grep -qE '\-X\s+(POST|PUT|PATCH|DELETE)' && return 0
  # Data flags (curl)
  echo "$cmd" | grep -qE '(\s|^)-(d|F)\s|--data\b|--data-raw\b|--data-binary\b|--data-urlencode\b|--form\b|--upload-file\b' && return 0
  # Pipe-to-curl pattern (data from stdin)
  echo "$cmd" | grep -qE '\|\s*curl\b' && return 0
  return 1
}
```

**is_webfetch_payload() function:** Returns 0 if WebFetch has a non-GET method.
```bash
is_webfetch_payload() {
  local method
  method=$(echo "$INPUT" | jq -r '.tool_input.method // "GET"' | tr '[:lower:]' '[:upper:]')
  [[ "$method" != "GET" && "$method" != "HEAD" ]]
}
```

**domain_matches() function:** Subdomain matching per D-05.
```bash
domain_matches() {
  local domain="$1"
  local entry="$2"
  [ "$domain" = "$entry" ] && return 0
  [[ "$domain" == *".$entry" ]] && return 0
  return 1
}
```

**domain_in_whitelist() function:** Checks if domain matches any entry in secrets[].allowed_domains.
```bash
domain_in_whitelist() {
  local domain="$1"
  local entries
  entries=$(jq -r '.secrets[].allowed_domains[]' "$WHITELIST" 2>/dev/null)
  while IFS= read -r entry; do
    [ -z "$entry" ] && continue
    domain_matches "$domain" "$entry" && return 0
  done <<< "$entries"
  return 1
}
```

**domain_in_readonly() function:** Checks if domain matches any entry in readonly_domains.
```bash
domain_in_readonly() {
  local domain="$1"
  local entries
  entries=$(jq -r '.readonly_domains[]' "$WHITELIST" 2>/dev/null)
  while IFS= read -r entry; do
    [ -z "$entry" ] && continue
    domain_matches "$domain" "$entry" && return 0
  done <<< "$entries"
  return 1
}
```

**register_call_id() function:** Generates UUID, registers with validator, returns 0 on success.
```bash
register_call_id() {
  local domain="$1"
  local call_id
  call_id=$(uuidgen)
  local response
  response=$(curl -s -w "\n%{http_code}" -X POST "${VALIDATOR_URL}/register" \
    -H "Content-Type: application/json" \
    -d "{\"call_id\": \"${call_id}\", \"domain\": \"${domain}\"}" 2>&1)
  local http_code
  http_code=$(echo "$response" | tail -1)
  local body
  body=$(echo "$response" | head -1)
  if [ "$http_code" = "200" ]; then
    log "REGISTERED call-id=${call_id} domain=${domain}"
    return 0
  else
    log "REGISTER FAILED: http_code=${http_code} body=${body}"
    return 1
  fi
}
```

**Main logic (decision tree per research Pattern 1):**

```bash
# Non-network Bash commands (no curl/wget) -- allow
if [ "$TOOL_NAME" = "Bash" ]; then
  cmd=$(echo "$INPUT" | jq -r '.tool_input.command // empty')
  if ! echo "$cmd" | grep -qE '\bcurl\b|\bwget\b'; then
    allow "non-network Bash command"
  fi
fi

# WebSearch -- always allow (read-only search, per CALL-04)
if [ "$TOOL_NAME" = "WebSearch" ]; then
  allow "WebSearch is read-only"
fi

# Extract domain
DOMAIN=$(extract_domain)

# Obfuscation detected
if [ "$DOMAIN" = "__OBFUSCATED__" ]; then
  deny "Blocked: potential URL obfuscation detected in command (fail-closed per security policy)"
fi

# No URL found in curl/wget command
if [ "$DOMAIN" = "__NO_URL__" ]; then
  deny "Blocked: curl/wget command but no URL could be extracted (fail-closed per security policy)"
fi

# Empty domain (shouldn't happen for WebFetch, but safety check)
if [ -z "$DOMAIN" ]; then
  allow "no target domain detected"
fi

# Determine if this is a payload (outbound data) request
IS_PAYLOAD=false
if [ "$TOOL_NAME" = "Bash" ]; then
  has_payload && IS_PAYLOAD=true
elif [ "$TOOL_NAME" = "WebFetch" ]; then
  is_webfetch_payload && IS_PAYLOAD=true
fi

# PAYLOAD requests: must go to whitelisted domain + register call-ID
if [ "$IS_PAYLOAD" = "true" ]; then
  if domain_in_whitelist "$DOMAIN"; then
    if register_call_id "$DOMAIN"; then
      allow "payload to whitelisted domain ${DOMAIN} (call-ID registered)"
    else
      deny "Blocked: failed to register call-ID with validator for ${DOMAIN}"
    fi
  else
    deny "Blocked: outbound payload to non-whitelisted domain ${DOMAIN}"
  fi
else
  # READ-ONLY (GET) requests: allow to any domain without registration (per CALL-04/D-08)
  allow "read-only request to ${DOMAIN} (no registration needed)"
fi
```
