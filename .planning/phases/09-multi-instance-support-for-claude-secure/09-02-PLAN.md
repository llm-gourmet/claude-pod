```bash
migrate_if_needed() {
  if [ -f "$CONFIG_DIR/.env" ] && [ ! -d "$CONFIG_DIR/instances" ]; then
    echo "Migrating single-instance config to multi-instance layout..."
    mkdir -p "$CONFIG_DIR/instances/default"
    mv "$CONFIG_DIR/.env" "$CONFIG_DIR/instances/default/.env"
    # Copy whitelist from app source (it was a docker mount, not stored in CONFIG_DIR)
    cp "$APP_DIR/config/whitelist.json" "$CONFIG_DIR/instances/default/whitelist.json"
    # Extract WORKSPACE_PATH into instance config, keep APP_DIR+PLATFORM in global
    grep '^WORKSPACE_PATH=' "$CONFIG_DIR/config.sh" > "$CONFIG_DIR/instances/default/config.sh" || true
    local tmp=$(mktemp)
    grep -v '^WORKSPACE_PATH=' "$CONFIG_DIR/config.sh" > "$tmp"
    mv "$tmp" "$CONFIG_DIR/config.sh"
    echo "Migration complete. Your existing setup is now instance 'default'."
    echo "Use: claude-secure --instance default"
  fi
}
```

**2. Parse --instance flag (per D-01, D-03):**

```bash
# Parse --instance before anything else
INSTANCE=""
RAW_ARGS=("$@")
ARGS=()
while [[ $# -gt 0 ]]; do
  case "$1" in
    --instance)
      if [ -z "${2:-}" ]; then
        echo "ERROR: --instance requires a name" >&2; exit 1
      fi
      INSTANCE="$2"; shift 2 ;;
    *) ARGS+=("$1"); shift ;;
  esac
done

# list and help don't need --instance
FIRST_ARG="${ARGS[0]:-}"
if [ "$FIRST_ARG" != "list" ] && [ "$FIRST_ARG" != "help" ] && [ "$FIRST_ARG" != "--help" ] && [ "$FIRST_ARG" != "-h" ]; then
  if [ -z "$INSTANCE" ]; then
    echo "ERROR: --instance NAME is required for all commands except 'list' and 'help'" >&2
    echo "Usage: claude-secure --instance NAME [command] [flags]" >&2
    exit 1
  fi
fi
```

**3. Validate instance name (per D-02):**

```bash
validate_instance_name() {
  local name="$1"
  if [[ ! "$name" =~ ^[a-z0-9][a-z0-9-]*$ ]]; then
    echo "ERROR: Instance name must be DNS-safe (lowercase alphanumeric and hyphens, must start with letter or digit)" >&2
    exit 1
  fi
  if [ ${#name} -gt 63 ]; then
    echo "ERROR: Instance name must be 63 characters or less" >&2
    exit 1
  fi
}

if [ -n "$INSTANCE" ]; then
```

**4. Auto-create instance on first use (per D-09):**

```bash
INSTANCE_DIR="$CONFIG_DIR/instances/$INSTANCE"

create_instance() {
  local name="$1"
  local idir="$CONFIG_DIR/instances/$name"
  echo "Instance '$name' not found. Creating..."
  read -rp "Workspace path [$HOME/claude-workspace-$name]: " ws_path
  ws_path="${ws_path:-$HOME/claude-workspace-$name}"
  ws_path="$(realpath -m "$ws_path")"
  mkdir -p "$ws_path"
  mkdir -p "$idir"
  echo "WORKSPACE_PATH=\"$ws_path\"" > "$idir/config.sh"
  # Copy whitelist template from app
  cp "$APP_DIR/config/whitelist.json" "$idir/whitelist.json"
  # Copy auth from existing instance or prompt
  local existing_env=""
  for d in "$CONFIG_DIR/instances"/*/; do
    [ -f "$d/.env" ] && existing_env="$d/.env" && break
  done
  if [ -n "$existing_env" ]; then
    read -rp "Copy auth credentials from $(basename "$(dirname "$existing_env")")? [Y/n]: " copy_auth
    if [[ "${copy_auth:-y}" =~ ^[Yy]$ ]]; then
      cp "$existing_env" "$idir/.env"
      echo "Auth credentials copied. Edit $idir/.env to customize secrets."
    else
      setup_instance_auth "$idir"
    fi
  else
    setup_instance_auth "$idir"
  fi
  echo "Instance '$name' created."
}

setup_instance_auth() {
  local idir="$1"
  echo "Choose authentication method:"
  echo "  1) OAuth token [recommended]"
  echo "  2) API key"
  read -rp "Choice [1]: " auth_choice
  auth_choice="${auth_choice:-1}"
  case "$auth_choice" in
    2)
      read -rsp "API key: " key; echo ""
      echo "ANTHROPIC_API_KEY=${key}" > "$idir/.env"
      ;;
    *)
      echo "Run 'claude setup-token' first to get your OAuth token."
      read -rsp "OAuth token: " token; echo ""
      echo "CLAUDE_CODE_OAUTH_TOKEN=${token}" > "$idir/.env"
      ;;
  esac
  echo "" >> "$idir/.env"
  echo "# Add secrets below (must match env_var in whitelist.json)" >> "$idir/.env"
  echo "# Example: GITHUB_TOKEN=ghp_your_token_here" >> "$idir/.env"
  chmod 600 "$idir/.env"
}

# Auto-create if instance specified but doesn't exist (skip for stop/remove/list/help)
if [ -n "$INSTANCE" ] && [ ! -d "$INSTANCE_DIR" ]; then
  if [ "$FIRST_ARG" = "stop" ] || [ "$FIRST_ARG" = "remove" ]; then
    echo "ERROR: Instance '$INSTANCE' does not exist." >&2
    exit 1
  fi
  create_instance "$INSTANCE"
fi
```

**8. Add `list` subcommand (per D-08, D-10):**

Inside the case statement, add before the help case:

```bash
  list)
    printf "%-20s %-10s %s\n" "INSTANCE" "STATUS" "WORKSPACE"
    printf "%-20s %-10s %s\n" "--------" "------" "---------"
    # Get running compose projects
    declare -A running_projects
    while IFS= read -r line; do
      name=$(echo "$line" | jq -r '.Name // empty' 2>/dev/null)
      [ -n "$name" ] && running_projects["$name"]=1
    done < <(docker compose ls --format json 2>/dev/null | jq -c '.[]?' 2>/dev/null)
    # Show all configured instances
    if [ -d "$CONFIG_DIR/instances" ]; then
      for idir in "$CONFIG_DIR/instances"/*/; do
        [ -d "$idir" ] || continue
        name=$(basename "$idir")
        ws=$(grep '^WORKSPACE_PATH=' "$idir/config.sh" 2>/dev/null | cut -d'"' -f2)
        project="claude-${name}"
        if [ -n "${running_projects[$project]:-}" ]; then
          status="running"
        else
          status="stopped"
        fi
        printf "%-20s %-10s %s\n" "$name" "$status" "$ws"
      done
    else
      echo "No instances configured."
    fi
    ;;
```

**9. Add `remove` subcommand:**

```bash
  remove)
    if [ -z "$INSTANCE" ]; then
      echo "ERROR: --instance NAME is required for remove" >&2; exit 1
    fi
    echo "This will stop containers and delete config for instance '$INSTANCE'."
    read -rp "Are you sure? [y/N]: " confirm
    if [[ ! "$confirm" =~ ^[Yy]$ ]]; then
      echo "Cancelled."; exit 0
    fi
    # Stop containers
    docker compose down --remove-orphans 2>/dev/null || true
    # Remove instance config
    rm -rf "$INSTANCE_DIR"
    echo "Instance '$INSTANCE' removed."
    echo "Note: Docker volumes (workspace, validator-db) are preserved."
    echo "Run 'docker volume ls | grep claude-${INSTANCE}' to see them."
    ;;
```

```bash
  logs)
```
