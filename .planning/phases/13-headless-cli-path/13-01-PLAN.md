spawn_project_name() {
  local profile="$1"
  local uuid8

}

```bash
  spawn)
    # Per D-04: --profile is required for spawn (enforced above by profile resolution)
    if [ -z "$PROFILE" ]; then
      echo "ERROR: --profile is required for spawn" >&2
      echo "Usage: claude-secure spawn --profile <name> --event '<json>'" >&2
      exit 1
    fi

    # Parse spawn-specific flags from REMAINING_ARGS (skip first which is "spawn")
    EVENT_JSON=""
    EVENT_FILE=""
    PROMPT_TEMPLATE=""
    DRY_RUN=0
    local i=1
    while [ $i -lt ${#REMAINING_ARGS[@]} ]; do
      case "${REMAINING_ARGS[$i]:-}" in
        --event)          EVENT_JSON="${REMAINING_ARGS[$((i+1))]:-}"; i=$((i+2)) ;;
        --event-file)     EVENT_FILE="${REMAINING_ARGS[$((i+1))]:-}"; i=$((i+2)) ;;
        --prompt-template) PROMPT_TEMPLATE="${REMAINING_ARGS[$((i+1))]:-}"; i=$((i+2)) ;;
        --dry-run)        DRY_RUN=1; i=$((i+1)) ;;
        *)                i=$((i+1)) ;;
      esac
    done

    # Load event from file if --event-file used (per D-02)
    if [ -n "$EVENT_FILE" ] && [ -z "$EVENT_JSON" ]; then
      if [ ! -f "$EVENT_FILE" ]; then
        echo "ERROR: Event file not found: $EVENT_FILE" >&2
        exit 1
      fi
      EVENT_JSON=$(cat "$EVENT_FILE")
    fi

    # Validate event JSON is provided and valid
    if [ -z "$EVENT_JSON" ]; then
      echo "ERROR: --event or --event-file is required for spawn" >&2
      echo "Usage: claude-secure spawn --profile <name> --event '<json>'" >&2
      exit 1
    fi
    if ! echo "$EVENT_JSON" | jq empty 2>/dev/null; then
      echo "ERROR: Invalid JSON in --event" >&2
      exit 1
    fi

    # Generate ephemeral project name (per D-05)
    COMPOSE_PROJECT_NAME=$(spawn_project_name "$PROFILE")
    export COMPOSE_PROJECT_NAME

    # Set cleanup trap (per D-07) - overrides the default cleanup trap
    trap spawn_cleanup EXIT

    # Read max_turns from profile.json (per D-08)
    MAX_TURNS=$(jq -r '.max_turns // empty' "$CONFIG_DIR/profiles/$PROFILE/profile.json")

    # TODO: Plan 02 implements template resolution, execution, and output capture
    # TODO: Plan 03 implements prompt template resolution and rendering

    echo "ERROR: spawn execution not yet implemented (pending Plans 02/03)" >&2
    exit 1
    ;;
```
