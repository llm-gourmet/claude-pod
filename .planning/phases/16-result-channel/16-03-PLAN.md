    === Helper 1: _extract_result_text_to_tempfile ===

    Based on Pitfall 4 + D-16 (16KB limit, 2x the 8192 default). `extract_payload_field` already has the hygiene + truncation logic at lines 438-450 but its LIMIT is hardcoded at 8192. Rather than parameterizing the existing helper (risky — Phase 15 tests depend on 8192), write a dedicated variant that inlines the python3 block with LIMIT=16384:

    ```bash
    # Phase 16 D-16 / Pitfall 4: extract .claude.result from the envelope and
    # emit the UTF-8-safe, control-char-stripped, 16KB-truncated bytes to a
    # fresh tempfile. Dedicated variant (vs extract_payload_field) so the
    # Phase 15 8KB payload limit stays unchanged for prompt rendering.
    _extract_result_text_to_tempfile() {
      local envelope_source="$1"
      local out_file
      out_file=$(mktemp)
      _CLEANUP_FILES+=("$out_file")

      local raw
      if [ -f "$envelope_source" ]; then
        raw=$(jq -r '.claude.result // empty' "$envelope_source" 2>/dev/null)
      else
        raw=$(printf '%s' "$envelope_source" | jq -r '.claude.result // empty' 2>/dev/null)
      fi

      RAW_PAYLOAD="$raw" python3 -c '
    import os, sys
    data = os.environb.get(b"RAW_PAYLOAD", b"")
    keep = {0x09, 0x0A, 0x0D}
    cleaned = bytes(b for b in data if b >= 0x20 or b in keep)
    LIMIT = 16384
    if len(cleaned) > LIMIT:
        truncated = cleaned[:LIMIT].decode("utf-8", errors="ignore").encode("utf-8")
        dropped = len(cleaned) - len(truncated)
        suffix = f"... [truncated {dropped} more bytes]".encode("utf-8")
        cleaned = truncated + suffix
    sys.stdout.buffer.write(cleaned)
    ' > "$out_file"

      echo "$out_file"
    }
    ```

    === Helper 2: render_report_template ===

    Based on Pattern G and Pitfall 2 (RESULT_TEXT/ERROR_MESSAGE LAST):

    ```bash
    # Phase 16 D-10 / D-11 / Pattern G / Pitfall 2: render a report template.
    # Chains on top of Phase 15's render_template (which substitutes all D-16
    # variables using the awk-from-file pattern) then substitutes the six
    # D-10 extension variables. RESULT_TEXT and ERROR_MESSAGE are deliberately
    # LAST so embedded {{ISSUE_TITLE}} style tokens inside Claude's output
    # are not re-interpreted as template variables (Pitfall 2).
    render_report_template() {
      local template_path="$1"
      local event_json="$2"
      local envelope_json="$3"
      local status="$4"

      # Step 1: run Phase 15 render_template to substitute all D-16 variables
      local rendered
      rendered=$(render_template "$template_path" "$event_json")
      local v_file

      # Step 2: substitute D-10 metadata variables (short single-line) FIRST,
      # so that if RESULT_TEXT contains e.g. {{COST_USD}}, the inner token
      # survives the RESULT_TEXT substitution (which runs last).
      #
      # Pitfall 5: cost_usd // cost and duration_ms // duration fallback
      # because Claude Code has shipped both field names historically.
      v_file=$(_extract_to_tempfile "$envelope_json" '.claude.cost_usd // .claude.cost // 0 | tostring' "0")
      rendered=$(_substitute_token_from_file "$rendered" "COST_USD" "$v_file")

      v_file=$(_extract_to_tempfile "$envelope_json" '.claude.duration_ms // .claude.duration // 0 | tostring' "0")
      rendered=$(_substitute_token_from_file "$rendered" "DURATION_MS" "$v_file")

      v_file=$(_extract_to_tempfile "$envelope_json" '.claude.session_id // ""' "")
      rendered=$(_substitute_token_from_file "$rendered" "SESSION_ID" "$v_file")

      # Wrapper-generated: TIMESTAMP and STATUS come from the caller, not the envelope.
      v_file=$(mktemp); _CLEANUP_FILES+=("$v_file")
      date -u +"%Y-%m-%dT%H:%M:%SZ" > "$v_file"
      rendered=$(_substitute_token_from_file "$rendered" "TIMESTAMP" "$v_file")

      v_file=$(mktemp); _CLEANUP_FILES+=("$v_file")
      printf '%s' "$status" > "$v_file"
      rendered=$(_substitute_token_from_file "$rendered" "STATUS" "$v_file")

      # Step 3: LAST — RESULT_TEXT and ERROR_MESSAGE.
      # These carry arbitrary Claude output that may contain literal {{TOKEN}}
      # strings. Substituting them AFTER all other passes guarantees those
      # literal tokens are preserved (Pitfall 2).
      v_file=$(_extract_result_text_to_tempfile "$envelope_json")
      rendered=$(_substitute_multiline_token_from_file "$rendered" "RESULT_TEXT" "$v_file")

      v_file=$(_extract_to_tempfile "$envelope_json" '.error // ""' "")
      rendered=$(_substitute_multiline_token_from_file "$rendered" "ERROR_MESSAGE" "$v_file")

      printf '%s' "$rendered"
    }
    ```

    === Helper 3: redact_report_file ===

    Copy Pattern B VERBATIM. Paste the full function body without modification — it was designed for this exact purpose and has the empty-value skip + metacharacter-safe behavior baked in:

    ```bash
    # Phase 16 D-15 / Pattern B / Pitfall 1: redact every non-empty profile
    # .env value from a rendered report file in place. Uses awk-from-file
    # LITERAL substring replace (index + substr) so any value metacharacter
    # (| / \ & $ [ ] ^ * .) is safe. NEVER uses sed with interpolation.
    redact_report_file() {
      local report_file="$1"
      local env_file="$2"
      local tmp_out tmp_val
      tmp_out=$(mktemp)
      tmp_val=$(mktemp)
      _CLEANUP_FILES+=("$tmp_out" "$tmp_val")

      while IFS='=' read -r raw_key raw_val || [ -n "$raw_key" ]; do
        [[ -z "$raw_key" || "$raw_key" =~ ^[[:space:]]*# ]] && continue
        raw_key="${raw_key#export }"
        raw_key="${raw_key//[[:space:]]/}"
        [ -z "$raw_key" ] && continue
        raw_val="${raw_val%$'\r'}"
        raw_val="${raw_val#\"}"; raw_val="${raw_val%\"}"
        raw_val="${raw_val#\'}"; raw_val="${raw_val%\'}"
        # D-15: skip empty values
        [ -z "$raw_val" ] && continue

        printf '%s' "$raw_val" > "$tmp_val"

        awk -v repl="<REDACTED:${raw_key}>" -v vfile="$tmp_val" '
          BEGIN {
            value = ""
            first = 1
            while ((getline line < vfile) > 0) {
              if (first) { value = line; first = 0 }
              else       { value = value "\n" line }
            }
            close(vfile)
            if (value == "") exit 0
          }
          {
            line = $0
            out = ""
            while ((i = index(line, value)) > 0) {
              out = out substr(line, 1, i-1) repl
              line = substr(line, i + length(value))
            }
            print out line
          }
        ' "$report_file" > "$tmp_out"
        mv "$tmp_out" "$report_file"
        : > "$tmp_val"
      done < "$env_file"

      rm -f "$tmp_val" 2>/dev/null || true
    }
    ```

    === Helper 4: write_audit_entry ===

    ```bash
    # Phase 16 D-04 / D-06 / D-07 / Pattern C / Pitfall 8 / Pitfall 14:
    # Append a single audit line to $LOG_DIR/${LOG_PREFIX}executions.jsonl.
    # Uses jq -cn for safe JSON construction. Enforces a 4095-byte guard so
    # POSIX O_APPEND atomicity (< PIPE_BUF = 4096 on Linux) is preserved under
    # concurrent same-profile spawns (Phase 14 semaphore up to 3).
    write_audit_entry() {
      local ts="$1" delivery_id="$2" webhook_id="$3" event_type="$4"
      local profile="$5" repo="$6" commit_sha="$7" branch="$8"
      local cost_usd="$9" duration_ms="${10}" session_id="${11}"
      local status="${12}" report_url="${13}" error_short="${14:-}"

      mkdir -p "$LOG_DIR"
      local audit_file="$LOG_DIR/${LOG_PREFIX}executions.jsonl"

      # jq_cost / jq_duration normalize: "" -> null, numeric -> number.
      local jq_cost="null" jq_duration="null"
      [ -n "$cost_usd" ] && jq_cost="$cost_usd"
      [ -n "$duration_ms" ] && jq_duration="$duration_ms"

      local line
      line=$(jq -cn \
        --arg ts "$ts" \
        --arg delivery_id "$delivery_id" \
        --arg webhook_id "$webhook_id" \
        --arg event_type "$event_type" \
        --arg profile "$profile" \
        --arg repo "$repo" \
        --arg commit_sha "$commit_sha" \
        --arg branch "$branch" \
        --arg session_id "$session_id" \
        --arg status "$status" \
        --arg report_url "$report_url" \
        --arg error_short "$error_short" \
        --argjson cost_usd "$jq_cost" \
        --argjson duration_ms "$jq_duration" \
        '{
          ts: $ts,
          delivery_id: $delivery_id,
          webhook_id: (if $webhook_id == "" then null else $webhook_id end),
          event_type: $event_type,
          profile: $profile,
          repo: (if $repo == "" then null else $repo end),
          commit_sha: (if $commit_sha == "" then null else $commit_sha end),
          branch: (if $branch == "" then null else $branch end),
          cost_usd: $cost_usd,
          duration_ms: $duration_ms,
          session_id: (if $session_id == "" then null else $session_id end),
          status: $status,
          report_url: (if $report_url == "" then null else $report_url end),
          error_short: (if $error_short == "" then null else $error_short end)
        }')

      # Pattern C / Pitfall 14: refuse to emit over-budget lines; drop error_short
      # and retry once. If still over, record a sentinel and proceed.
      if [ "${#line}" -gt 4095 ]; then
        echo "WARNING: audit line exceeds PIPE_BUF, dropping error_short" >&2
        line=$(jq -cn \
          --arg ts "$ts" --arg delivery_id "$delivery_id" --arg webhook_id "$webhook_id" \
          --arg event_type "$event_type" --arg profile "$profile" --arg repo "$repo" \
          --arg commit_sha "$commit_sha" --arg branch "$branch" \
          --arg session_id "$session_id" --arg status "$status" --arg report_url "$report_url" \
          --argjson cost_usd "$jq_cost" --argjson duration_ms "$jq_duration" \
          '{
            ts: $ts, delivery_id: $delivery_id,
            webhook_id: (if $webhook_id == "" then null else $webhook_id end),
            event_type: $event_type, profile: $profile,
            repo: (if $repo == "" then null else $repo end),
            commit_sha: (if $commit_sha == "" then null else $commit_sha end),
            branch: (if $branch == "" then null else $branch end),
            cost_usd: $cost_usd, duration_ms: $duration_ms,
            session_id: (if $session_id == "" then null else $session_id end),
            status: $status,
            report_url: (if $report_url == "" then null else $report_url end),
            error_short: "<dropped: over-budget>"
          }')
      fi

      printf '%s\n' "$line" >> "$audit_file"
    }
    ```

    === Helper 5: push_with_retry ===

    Based on Pattern D. Verbatim with LC_ALL=C added:

    ```bash
    # Phase 16 D-14 / Pattern D / Pitfall 6: push the currently-committed
    # report to origin/$branch. On non-fast-forward rejection, rebase once
    # and retry exactly once. Never force-push. "Everything up-to-date" is
    # a success (exit 0), even after a rebase where our commit was already
    # upstream. LC_ALL=C makes git's English error messages deterministic
    # for the retry-trigger grep.
    push_with_retry() {
      local clone_dir="$1" branch="$2"
      local repo_path="$clone_dir/repo"

      local err_log="$clone_dir/push.err"

      if LC_ALL=C GIT_ASKPASS="$clone_dir/.askpass.sh" \
         GIT_ASKPASS_PAT="$REPORT_REPO_TOKEN" \
         GIT_TERMINAL_PROMPT=0 \
         git -C "$repo_path" -c credential.helper= -c credential.helper='' \
             push origin "$branch" 2>"$err_log"; then
        return 0
      fi

      if grep -q -e 'non-fast-forward' -e 'Updates were rejected' "$err_log"; then
        echo "INFO: report push rejected (non-fast-forward). Rebasing and retrying once." >&2
        if ! LC_ALL=C GIT_ASKPASS="$clone_dir/.askpass.sh" \
             GIT_ASKPASS_PAT="$REPORT_REPO_TOKEN" \
             GIT_TERMINAL_PROMPT=0 \
             git -C "$repo_path" -c credential.helper= -c credential.helper='' \
                 pull --rebase origin "$branch" 2>>"$err_log"; then
          sed "s|${REPORT_REPO_TOKEN}|<REDACTED:REPORT_REPO_TOKEN>|g" "$err_log" >&2
          return 1
        fi
        if LC_ALL=C GIT_ASKPASS="$clone_dir/.askpass.sh" \
           GIT_ASKPASS_PAT="$REPORT_REPO_TOKEN" \
           GIT_TERMINAL_PROMPT=0 \
           git -C "$repo_path" -c credential.helper= -c credential.helper='' \
               push origin "$branch" 2>>"$err_log"; then
          return 0
        fi
      fi

      sed "s|${REPORT_REPO_TOKEN}|<REDACTED:REPORT_REPO_TOKEN>|g" "$err_log" >&2
      return 1
    }
    ```

    === Helper 6: publish_report ===

    Based on Pattern A extended with commit + push integration. The signature takes the ALREADY-RENDERED, ALREADY-REDACTED report body as a path:

    ```bash
    # Phase 16 D-01 .. D-03 / D-12 / D-13 / D-14 / Pattern A / Pitfall 3 / Pitfall 11:
    # Publish a rendered+redacted report file to the configured doc repo via
    # fresh shallow clone + commit + push (retry once on non-fast-forward).
    #
    # Args:
    #   $1 = rendered_redacted_body_path
    #   $2 = event_type
    #   $3 = delivery_id
    #   $4 = delivery_id_short (last 8 hex chars, Open Question 2)
    #   $5 = repo_full_name (e.g. "owner/name", or "" if unknown)
    #
    # Returns:
    #   exit 0  + stdout last line = report_url (on push success)
    #   exit 2  (skip; REPORT_REPO or REPORT_REPO_TOKEN empty)
    #   exit 1  (push failure)
    publish_report() {
      local body_path="$1"
      local event_type="$2"
      local delivery_id="$3"
      local delivery_id_short="$4"
      local repo_full_name="$5"

      local repo_url="${REPORT_REPO:-}"
      local branch="${REPORT_BRANCH:-main}"
      local prefix="${REPORT_PATH_PREFIX:-reports}"
      local pat="${REPORT_REPO_TOKEN:-}"

      if [ -z "$repo_url" ] || [ -z "$pat" ]; then
        return 2
      fi

      local clone_dir
      clone_dir=$(mktemp -d "${TMPDIR:-/tmp}/cs-report-XXXXXXXX")
      _CLEANUP_FILES+=("$clone_dir")

      # Ephemeral askpass helper (Pattern A). Lives inside clone_dir so
      # spawn_cleanup wipes it along with the clone.
      local askpass="$clone_dir/.askpass.sh"
      cat > "$askpass" <<'ASKPASS'
    #!/bin/sh
    case "$1" in
      Username*) printf 'x-access-token\n' ;;
      Password*) printf '%s\n' "$GIT_ASKPASS_PAT" ;;
    esac
    ASKPASS
      chmod 700 "$askpass"

      # Pitfall 3: pipe git stderr through sed to scrub any accidental PAT echo.
      # Pitfall 11: bounded clone via GIT_HTTP_LOW_SPEED_* + outer `timeout 60`.
      if ! LC_ALL=C \
           GIT_ASKPASS="$askpass" \
           GIT_ASKPASS_PAT="$pat" \
           GIT_TERMINAL_PROMPT=0 \
           GIT_HTTP_LOW_SPEED_LIMIT=1 \
           GIT_HTTP_LOW_SPEED_TIME=30 \
           timeout 60 \
           git -c credential.helper= -c credential.helper='' \
               -c core.autocrlf=false \
               clone --depth 1 --branch "$branch" --quiet "$repo_url" "$clone_dir/repo" \
               2> >(sed "s|${pat}|<REDACTED:REPORT_REPO_TOKEN>|g" >&2); then
        return 1
      fi

      # D-12: place file at <prefix>/<YYYY>/<MM>/<event_type>-<id8>.md
      local year month
      year=$(date -u +%Y)
      month=$(date -u +%m)
      local rel_path="${prefix}/${year}/${month}/${event_type}-${delivery_id_short}.md"
      local abs_path="$clone_dir/repo/$rel_path"
      mkdir -p "$(dirname "$abs_path")"
      cp "$body_path" "$abs_path"

      # D-13: authored via env vars (no host git config touched).
      local commit_repo="${repo_full_name:-unknown}"
      local commit_msg="report(${event_type}): ${commit_repo} ${delivery_id_short}"

      if ! LC_ALL=C \
           GIT_AUTHOR_NAME="claude-secure" \
           GIT_AUTHOR_EMAIL="claude-secure@localhost" \
           GIT_COMMITTER_NAME="claude-secure" \
           GIT_COMMITTER_EMAIL="claude-secure@localhost" \
           git -C "$clone_dir/repo" -c core.autocrlf=false add "$rel_path"; then
        return 1
      fi

      if ! LC_ALL=C \
           GIT_AUTHOR_NAME="claude-secure" \
           GIT_AUTHOR_EMAIL="claude-secure@localhost" \
           GIT_COMMITTER_NAME="claude-secure" \
           GIT_COMMITTER_EMAIL="claude-secure@localhost" \
           git -C "$clone_dir/repo" -c core.autocrlf=false commit -q -m "$commit_msg"; then
        # nothing to commit (e.g. identical file on replay) -> treat as success,
        # emit the expected URL anyway so audit records the known location.
        :
      fi

      if ! push_with_retry "$clone_dir" "$branch"; then
        return 1
      fi

      # D-12 + D-14: derive report_url. For https URLs, map .git to /blob/<branch>/<rel>
      local url_base="$repo_url"
      url_base="${url_base%.git}"
      printf '%s/blob/%s/%s\n' "$url_base" "$branch" "$rel_path"
      return 0
    }
    ```

      body_file=$(mktemp); _CLEANUP_FILES+=("$body_file")

          echo "WARNING: redact_report_file failed; committing un-redacted is forbidden, aborting publish" >&2
      fi

        pub_out=$(publish_report "$body_file" "$event_type" "$delivery_id" "$delivery_id_short" "$audit_repo") && pub_rc=0 || pub_rc=$?
        case "${pub_rc:-0}" in
          0) report_url=$(printf '%s' "$pub_out" | tail -n 1); push_status="success" ;;
          2) push_status="success" ;;  # REPORT_REPO unset -> skip, not a failure
          *) push_status="report_push_failed" ;;
        esac
      fi
    fi

    audit_status="$push_status"
    write_audit_entry "$audit_ts" "$delivery_id" "$webhook_id" "$event_type" "$PROFILE" \
      "$audit_repo" "$audit_commit" "$audit_branch" \
      "$cost_usd" "$duration_ms" "$session_id" \
      "$audit_status" "$report_url" ""

    # D-18: always exit 0 on success regardless of push outcome.
    printf '%s\n' "$envelope_json"
    return 0
