# D-08: single-flight reaper cycle. Acquires a non-blocking flock on
# $LOG_DIR/${LOG_PREFIX}reaper.lock. If held (e.g., timer-driven cycle
# already running when operator invokes `claude-secure reap`), the second
# instance exits 0 silently with a journal log line.
do_reap() {
  local dry_run=0
  while [ $# -gt 0 ]; do
    case "$1" in
      --dry-run) dry_run=1; shift ;;
      --help|-h) echo "Usage: claude-secure reap [--dry-run]"; ...; return 0 ;;
      *) echo "reaper: unknown flag '$1'" >&2; return 2 ;;
    esac
  done
  export REAPER_DRY_RUN=$dry_run

  local lock_file="${LOG_DIR:-$CONFIG_DIR/logs}/${LOG_PREFIX:-}reaper.lock"
  mkdir -p "$(dirname "$lock_file")" 2>/dev/null || true
  exec 9>"$lock_file"
  chmod 666 "$lock_file" 2>/dev/null || true
  if ! flock -n 9; then
    echo "reaper: another instance is running (lock held), skipping cycle"
    return 0
  fi

  echo "reaper: cycle start prefix=${INSTANCE_PREFIX:-cs-}"
  REAPED_COUNT=0
  REAPED_ERRORS=0
  EVENTS_DELETED=0
  reap_orphan_projects "${INSTANCE_PREFIX:-cs-}"
  reap_stale_event_files
  echo "reaper: cycle end killed=${REAPED_COUNT} events_deleted=${EVENTS_DELETED} errors=${REAPED_ERRORS}"
  if [ "${REAPED_ERRORS:-0}" -gt 0 ] && [ "${REAPED_COUNT:-0}" -eq 0 ]; then
    return 1
  fi
  return 0
}
```

    do_reap() {
      local dry_run=0
      while [ $# -gt 0 ]; do
        case "$1" in
          --dry-run) dry_run=1; shift ;;
          --help|-h)
            echo "Usage: claude-secure reap [--dry-run]"
            echo "  Cleans up orphaned spawn containers and stale event files."
            return 0
            ;;
          *)
            echo "reaper: unknown flag '$1'" >&2
            echo "Usage: claude-secure reap [--dry-run]" >&2
            return 2
            ;;
        esac
      done
      export REAPER_DRY_RUN=$dry_run

      # Phase 18 PORT-03: mkdir-based atomic lock with PID file.
      local lockdir="${LOG_DIR:-$CONFIG_DIR/logs}/${LOG_PREFIX:-}reaper.lockdir"
      local pidfile="$lockdir/pid"
      mkdir -p "$(dirname "$lockdir")" 2>/dev/null || true

      if ! mkdir "$lockdir" 2>/dev/null; then
        # Lockdir already exists — check holder liveness
        local holder_pid=0
        if [ -r "$pidfile" ]; then
          holder_pid="$(cat "$pidfile" 2>/dev/null || echo 0)"
        fi
        if [ "${holder_pid:-0}" -gt 0 ] && kill -0 "$holder_pid" 2>/dev/null; then
          echo "reaper: another instance is running (pid=$holder_pid lockdir=$lockdir), skipping cycle"
          return 0
        fi
        # Stale: holder is dead. Reclaim.
        echo "reaper: stale lockdir found (dead holder pid=$holder_pid), reclaiming"
        rm -f "$pidfile" 2>/dev/null || true
        rmdir "$lockdir" 2>/dev/null || true
        if ! mkdir "$lockdir" 2>/dev/null; then
          echo "reaper: failed to acquire lockdir after reclaim attempt, skipping cycle"
          return 0
        fi
      fi
      echo "$$" > "$pidfile" 2>/dev/null || true
      # Cleanup on any exit path (normal return, error, signal)
      # shellcheck disable=SC2064
      trap "rm -f '$pidfile' 2>/dev/null || true; rmdir '$lockdir' 2>/dev/null || true" EXIT

      echo "reaper: cycle start prefix=${INSTANCE_PREFIX:-cs-}"

      REAPED_COUNT=0
      REAPED_ERRORS=0
      EVENTS_DELETED=0

      reap_orphan_projects "${INSTANCE_PREFIX:-cs-}"
      reap_stale_event_files

      echo "reaper: cycle end killed=${REAPED_COUNT} events_deleted=${EVENTS_DELETED} errors=${REAPED_ERRORS}"

      # D-10: whole-cycle failure (all projects errored, none reaped) flips exit code.
      if [ "${REAPED_ERRORS:-0}" -gt 0 ] && [ "${REAPED_COUNT:-0}" -eq 0 ]; then
        return 1
      fi
      return 0
    }
    ```
